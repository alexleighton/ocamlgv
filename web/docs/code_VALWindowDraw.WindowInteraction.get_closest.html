let get_closest g x y =
    match (get_one_vertex g) with
        Some v ->
          let aux v (best,d) =
            let vect = get_center v in
            let dist = (dist (x,y) (vect.x,vect.y)) in
              if d > dist then v,dist else best,d
          in
          let v,d = G.fold_vertex aux g (v,infinity) in Some v
      | None -> None
