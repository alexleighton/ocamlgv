let interactive (w,h,g) =
    try
      Graphics.open_graph "";
      let state = ref new_state in
      while true do
        ignore (DrawGraph.draw_image (w,h,g));
        let status = Graphics.wait_next_event events in
          state := handle_events g status !state
      done;
      g
    with
      | Failure "done" -> g
      | Graphic_failure msg -> g
